### Task 1: Add `setup` subcommand

**Files:**
- Modify: `bin/gcloud-auth-proxy`

**Step 1: Add the subcommand function**

Add `cmd_setup` to `bin/gcloud-auth-proxy` that:
- Writes `~/.config/ssh/config-gcloud-auth-proxy` containing:
  ```sshconfig
  # Generated by gcloud-auth-proxy setup
  Host *
    RemoteForward /tmp/gcloud-auth-proxy.sock <SOCKET_PATH>
  ```
  Where `<SOCKET_PATH>` is the resolved value of `$GCLOUD_AUTH_PROXY_SOCKET` (default `~/.local/state/gcloud-auth-proxy/proxy.sock`).
- Creates parent directory if needed
- Prints instructions: add `Include ~/.config/ssh/config-gcloud-auth-proxy` to your SSH config

```bash
cmd_setup() {
	local ssh_config_dir="${HOME}/.config/ssh"
	local ssh_config_path="${ssh_config_dir}/config-gcloud-auth-proxy"

	mkdir -p "$ssh_config_dir"

	cat >"$ssh_config_path" <<-EOF
	# Generated by gcloud-auth-proxy setup
	Host *
	  RemoteForward /tmp/gcloud-auth-proxy.sock ${SOCKET_PATH}
	EOF

	echo "SSH config written to: $ssh_config_path" >&2
	echo "" >&2
	echo "Add this line to your SSH config:" >&2
	echo "  Include ~/.config/ssh/config-gcloud-auth-proxy" >&2
}
```

### Task 2: Add `setup-remote` subcommand

**Files:**
- Modify: `bin/gcloud-auth-proxy`

**Step 1: Add the subcommand function**

Add `cmd_setup_remote` that takes an SSH host argument, constructs a setup script, and pipes it to `ssh <host> bash`:

```bash
cmd_setup_remote() {
	local host="${1:-}"

	if [[ -z "$host" ]]; then
		echo "Usage: gcloud-auth-proxy setup-remote <ssh-host>" >&2
		exit 1
	fi

	local remote_socket="/tmp/gcloud-auth-proxy.sock"
	local credential_helper_path="\$HOME/.config/gcloud/credential_helper.sh"

	ssh "$host" bash <<-'REMOTE_SCRIPT'
	set -euo pipefail

	SOCKET_PATH="/tmp/gcloud-auth-proxy.sock"
	CREDENTIAL_HELPER_PATH="$HOME/.config/gcloud/credential_helper.sh"

	mkdir -p "$(dirname "$CREDENTIAL_HELPER_PATH")"

	cat >"$CREDENTIAL_HELPER_PATH" <<'HELPER'
	#!/usr/bin/env bash
	set -euo pipefail

	SOCKET_PATH="/tmp/gcloud-auth-proxy.sock"

	if [[ ! -S "$SOCKET_PATH" ]]; then
	  echo "Error: gcloud-auth-proxy socket not found at $SOCKET_PATH" >&2
	  echo "Make sure SSH socket forwarding is active" >&2
	  exit 1
	fi

	token=$(curl -s --unix-socket "$SOCKET_PATH" http://localhost/token)

	if [[ -z "$token" ]]; then
	  echo "Error: Failed to get access token from proxy" >&2
	  exit 1
	fi

	cat <<JSON
	{
	  "access_token": "$token",
	  "token_expiry": "$(date -u -d '+1 hour' '+%Y-%m-%dT%H:%M:%SZ' 2>/dev/null || date -u -v+1H '+%Y-%m-%dT%H:%M:%SZ' 2>/dev/null || echo '')"
	}
	JSON
	HELPER

	chmod +x "$CREDENTIAL_HELPER_PATH"

	for rcfile in "$HOME/.bashrc" "$HOME/.zshrc"; do
	  if [[ -f "$rcfile" ]]; then
	    if ! grep -q "CLOUDSDK_AUTH_CREDENTIAL_FILE_OVERRIDE" "$rcfile"; then
	      printf '\n# gcloud auth proxy configuration\nexport CLOUDSDK_AUTH_CREDENTIAL_FILE_OVERRIDE="%s"\n' "$CREDENTIAL_HELPER_PATH" >>"$rcfile"
	    fi
	  fi
	done

	echo "Remote setup complete." >&2
	echo "  Credential helper: $CREDENTIAL_HELPER_PATH" >&2
	echo "  Reload your shell or run: source ~/.bashrc" >&2
	REMOTE_SCRIPT

	echo "Remote host '$host' configured." >&2
}
```
